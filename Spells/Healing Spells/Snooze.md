```
8009fe4c li     $v0, -0x0014 ; load -20
8009fe50 sh     $v0, 0x0004(v1) ; store -20 to the HP change
8009fe54 lui    $v1, 0x8014
8009fe58 lw     $v1, 0x63a0(v1)
8009fe5c li     $v0, -0x0004 ; load -4
8009fe60 lw     $ra, 0x0010(sp) ; store -4 to the AP change
8009fe64 sh     $v0, 0x0006(v1)
```
